156  std::vector<uint32_t> mechanism_literal_active() {
157 
158  auto item = *clause_banks.begin();
159  auto number_of_literals = item->number_of_literals;;
160 
161  size_t number_of_ta_chunks = (number_of_literals + 31) / 32; // Calculate the number of 32-bit chunks
162  std::vector<uint32_t> literal_active(number_of_ta_chunks, 0); // Initialize with zeros
163 
164  std::uniform_real_distribution<float> dist(0.0f, 1.0f); // Distribution for random floats between 0 and 1
165 
166  // Iterate over each literal to determine its activation based on the drop probability
167  for (size_t k = 0; k < number_of_literals; ++k) {
168 
169  auto rng = fast_rand() / static_cast<float>(UINT32_MAX);
170 
171  if (rng >= literal_drop_p) { // Literal remains active
172  size_t ta_chunk = k / 32;
173  size_t chunk_pos = k % 32;
174  literal_active[ta_chunk] |= (1 << chunk_pos);
175  }
176  }
177 
178  // If feature negation is not applied, clear the corresponding bits for the second half of literals
179  if (!feature_negation) {
180  for (size_t k = number_of_literals / 2; k < number_of_literals; ++k) {
181  size_t ta_chunk = k / 32;
182  size_t chunk_pos = k % 32;
183  literal_active[ta_chunk] &= ~(1 << chunk_pos);
184  }
185  }
186 
187  return literal_active;

209  float mechanism_compute_update_probabilities(bool is_target, int class_sum) {
210  // Confidence-driven updating method
211  if (confidence_driven_updating) {
