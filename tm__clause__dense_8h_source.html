179  void initializeClauses(){
180  // Set all bits to 1 except the last bit in each "chunk" of the 1D array
181  for (int i = 0; i < number_of_clauses; ++i) {
182  for (int j = 0; j < number_of_ta_chunks; ++j) {
183  for (int k = 0; k < number_of_state_bits - 1; ++k) { // Up to the second last
184  int index = i * (number_of_ta_chunks * number_of_state_bits) + j * number_of_state_bits + k;
185  clause_bank[index] = ~uint32_t(0); // Set all bits to 1
186  }
187  // Set the last bit to 0
188  int last_bit_index = i * (number_of_ta_chunks * number_of_state_bits) + j * number_of_state_bits + (number_of_state_bits - 1);
189  clause_bank[last_bit_index] = 0;
190  }
191  }
192 
193  // Sett all bits to 1 for independent clauses
194  std::fill(clause_bank_ind.begin(), clause_bank_ind.end(), ~0);
195  }

230  void setTAState(size_t clause, size_t ta, unsigned int state) {
231  size_t ta_chunk = ta / 32;
232  size_t chunk_pos = ta % 32;
233  size_t pos = calculatePosition(clause, ta_chunk);
234  for (size_t b = 0; b < number_of_state_bits; ++b) {
235  if (state & (1 << b)) {
236  clause_bank[pos + b] |= (1 << chunk_pos);
237  } else {
238  clause_bank[pos + b] &= ~(1 << chunk_pos);
239  }
240  }
241  }

243  unsigned int getTAState(size_t clause, size_t ta) {
244  size_t ta_chunk = ta / 32;
245  size_t chunk_pos = ta % 32;
246  size_t pos = calculatePosition(clause, ta_chunk);
247  unsigned int state = 0;
248  for (size_t b = 0; b < number_of_state_bits; ++b) {
249  if (clause_bank[pos + b] & (1 << chunk_pos)) {
250  state |= (1 << b);
251  }
252  }
253  return state;
254  }
