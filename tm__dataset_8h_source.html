56  static void generate_pattern_based_dataset(std::vector<std::vector<uint32_t>>& X, std::vector<uint32_t>& y, int num_samples) {
57  X.clear();
58  y.clear();
59  std::random_device rd;
60  std::mt19937 gen(rd());
61  std::uniform_int_distribution<> distrib(0, 1);
62 
63  for (int i = 0; i < num_samples; ++i) {
64  std::vector<uint32_t> x(50);
65  // Randomly generate binary features
66  for (auto& feature : x) {
67  feature = distrib(gen);
68  }
69 
70  // Determine the class based on a pattern in the features
71  // Here, we use the sum of '1's in each 10-feature group to influence the class
72  int sum[5] = {0};
73  for (int group = 0; group < 5; ++group) {
74  sum[group] = std::accumulate(x.begin() + group * 10, x.begin() + (group + 1) * 10, 0);
75  }
76 
77  // Example pattern-based class determination:
78  // Use the parity of the sum of '1's in each group to determine the class
79  uint32_t class_label = 0;
80  for (int group = 0; group < 5; ++group) {
81  if (sum[group] % 2 == 0) { // If the sum of '1's is even, add to the class label
82  class_label += 1 << group;
83  }
84  }
85 
86  // This scheme can produce 32 different classes (from 0 to 31) based on the combination of even/odd sums.
87  // To limit it to 10 classes, we can simply take the modulo 10 of the class_label.
88  class_label %= 10;
89 
90  X.push_back(x);
91  y.push_back(class_label);
92  }
